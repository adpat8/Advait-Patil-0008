package javaLogicTest;

import java.util.Scanner;

public class SegregateOddEven {
	
	public static int[] segregate(int[] arr)
	{
		int l = 0; int r = arr.length-1;
		int[] rarr = new int[arr.length];
		for(int i=0;i<arr.length;i++)
		{
			if(arr[i]%2!=0)
				rarr[r--]=arr[i];
			else
				rarr[l++]=arr[i];
		}
		return rarr;
	}

package javaLogicTest;

import java.util.Scanner;

public class BinarySearch {
	
	public static int[] sort(int[] arr)
	{
		int l=arr.length;
		for(int i=0;i<l;i++)
		{
			for(int j=0;j<l-1;j++)
			{
				if(arr[j]>arr[j+1])
				{
					int t=arr[j+1];
					arr[j+1]=arr[j];
					arr[j]=t;
				}
			}
		}
		return arr;
	}
	
	public static boolean search(int[] arr, int n)
	{
		int l=arr.length;
		arr=sort(arr);
		int s=0, e=l-1;
		boolean f=false;
		while(s<=e)
		{
			int m=(s+e)/2;
			if(n==arr[m])
			{
				f=true;
				break;
			}
			else if(n>arr[m])
				s=m+1;
			else
				e=m-1;
		}
		return f;
	}
	
	public static void main(String[] args) {
		
		//System.out.println("Enter the array size: ");
		Scanner sc=new Scanner(System.in);
		//int l = sc.nextInt();
		//int[] arr=new int[l];
		int l=5;
		int[] arr=new int[l];
		//System.out.println("Enter the elements of the array: ");
		System.out.println("Enter 5 numbers: ");
		for(int i=0;i<l;i++)
		{
			arr[i]=sc.nextInt();
		}
		System.out.println("Enter the element to search: ");
		int n = sc.nextInt();
		if(search(arr,n))
			System.out.println(n+" is present in the array.");
		else
			System.out.println(n+" is not present in the array.");
		
	}

}
